local function FNFZQ_fake_script() -- offlinetext.LocalScript 
	local script = Instance.new('LocalScript', offlinetext)

	local Players = game:GetService("Players")
	
	local un = Players.LocalPlayer.Name
	
	local function iterPageItems(pages)
		return coroutine.wrap(function()
			local pagenum = 1
			while true do
				for _, item in ipairs(pages:GetCurrentPage()) do
					coroutine.yield(item, pagenum)
				end
				if pages.IsFinished then
					break
				end
				pages:AdvanceToNextPageAsync()
				pagenum = pagenum + 1
			end
		end)
	end
	local userId = Players:GetUserIdFromNameAsync(un)
	local friendPages = Players:GetFriendsAsync(userId)
	local offline = {}
	for item, pageNo in iterPageItems(friendPages) do
		if item.IsOnline then
			continue
		else
			table.insert(offline, false)
		end
	end
	script.Parent.Text = "Offline: "..tostring(#offline)
end
coroutine.wrap(FNFZQ_fake_script)()
